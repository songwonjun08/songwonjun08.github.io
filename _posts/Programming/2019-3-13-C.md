## 기본 입출력

````c
// 숫자 입력 숫자 출력
#include <stdio.h>
int main (void) {
    
    int a;
   
    scanf("%d",&a);
    printf("%d",a);
    
    return 0;
}
````

````c
// 문자열 입력 출력
int main (void) {
    
    char str[50];
    
    // scanf는 stdin에서 ' '(공백), '\n', '\t'만나면 입력 종료
    // 따라서 공백 입력 받지 못함
    scanf("%s",str);
    
    // fgets(버퍼에 대한 포인터,문자열 최대 크기 '\0'포함,데이터 입력받을곳)
    fgets(str,sizeof(str),stdin);
    
    printf("%s",str);
    
    return 0;
}
````
